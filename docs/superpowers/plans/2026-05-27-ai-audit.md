# AI Audit Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an on-demand AI call audit feature that triggers Gemini to evaluate each AI participant's performance and stores the result.

**Architecture:** Five services are touched in dependency order: `bin-openapi-manager` (adds spec), `bin-dbscheme-manager` (adds table), `bin-ai-manager` (owns audit model + handler + Gemini logic), `bin-common-handler` (adds RPC client), `bin-api-manager` (HTTP layer). Each task is self-contained and produces working, verifiable changes.

**Tech Stack:** Go 1.21+, MySQL (squirrel query builder), Redis (no audit caching — audits are query-heavy), RabbitMQ RPC (sock pattern), Gin HTTP, go-openai SDK, Python Alembic (DB migrations), OpenAPI YAML (code-gen).

```go
package aiaudit

import (
    "encoding/json"
    "time"

    "github.com/gofrs/uuid"
    "monorepo/bin-common-handler/models/identity"
)

// WebhookMessage is the externally-visible representation of an AIAudit.
// Internal fields not listed here are stripped by ConvertWebhookMessage.
type WebhookMessage struct {
    identity.Identity

    AIcallID        uuid.UUID       `json:"aicall_id,omitempty"`
    AIID            uuid.UUID       `json:"ai_id,omitempty"`
    PromptHistoryID uuid.UUID       `json:"prompt_history_id,omitempty"`
    Status          Status          `json:"status,omitempty"`
    OverallScore    *int            `json:"overall_score"`
    Evaluation      json.RawMessage `json:"evaluation"`
    Language        string          `json:"language,omitempty"`
    Error           string          `json:"error,omitempty"`

    TMCreate *time.Time `json:"tm_create"`
    TMUpdate *time.Time `json:"tm_update"`
    TMDelete *time.Time `json:"tm_delete"`
}

// ConvertWebhookMessage converts an AIAudit to its webhook-safe representation.
func (a *AIAudit) ConvertWebhookMessage() *WebhookMessage {
    return &WebhookMessage{
        Identity:        a.Identity,
        AIcallID:        a.AIcallID,
        AIID:            a.AIID,
        PromptHistoryID: a.PromptHistoryID,
        Status:          a.Status,
        OverallScore:    a.OverallScore,
        Evaluation:      a.Evaluation,
        Language:        a.Language,
        Error:           a.Error,
        TMCreate:        a.TMCreate,
        TMUpdate:        a.TMUpdate,
        TMDelete:        a.TMDelete,
    }
}

// CreateWebhookEvent marshals the WebhookMessage to JSON.
func (a *AIAudit) CreateWebhookEvent() ([]byte, error) {
    return json.Marshal(a.ConvertWebhookMessage())
}
```

```go
package dbhandler

import (
    "context"
    "database/sql"
    "encoding/json"
    "fmt"

    sq "github.com/Masterminds/squirrel"
    uuid "github.com/gofrs/uuid"

    commondatabasehandler "monorepo/bin-common-handler/pkg/databasehandler"
    "monorepo/bin-ai-manager/models/aiaudit"
)

const (
    aiauditTable = "ai_ai_audits"
    aiauditSoftDeleteEpoch = "9999-01-01 00:00:00.000000"
)

// AIAuditUpsert inserts a new audit row or resets an existing non-progressing one.
// Returns rowsAffected from sql.Result:
//   - 0 means the row exists and is already 'progressing' (caller should return 409)
//   - >0 means the row was created or overwritten (caller should spawn goroutine)
//
// NOTE: The DB DSN must NOT include CLIENT_FOUND_ROWS. With that flag,
// MySQL returns 1 for no-op updates (rows matched, not rows changed),
// which would break the 0-means-in-progress detection.
func (h *handler) AIAuditUpsert(ctx context.Context, a *aiaudit.AIAudit) (int64, error) {
    a.TMCreate = h.utilHandler.TimeNow()
    a.TMUpdate = nil

    evaluationJSON, err := json.Marshal(a.Evaluation)
    if err != nil {
        return 0, fmt.Errorf("AIAuditUpsert: could not marshal evaluation. err: %v", err)
    }

    query := fmt.Sprintf(`
        INSERT INTO %s
            (id, customer_id, aicall_id, ai_id, prompt_history_id, status, overall_score, evaluation, language, error, tm_create, tm_update, tm_delete)
        VALUES
            (?, ?, ?, ?, ?, 'progressing', NULL, NULL, ?, NULL, ?, NULL, '%s')
        ON DUPLICATE KEY UPDATE
            status        = IF(status = 'progressing', status, 'progressing'),
            tm_delete     = '%s',
            overall_score = NULL,
            evaluation    = NULL,
            error         = NULL,
            language      = VALUES(language),
            prompt_history_id = VALUES(prompt_history_id),
            tm_update     = NULL
    `, aiauditTable, aiauditSoftDeleteEpoch, aiauditSoftDeleteEpoch)

    _ = evaluationJSON // evaluation starts NULL; used in UpdateFinal

    result, err := h.db.ExecContext(ctx, query,
        a.ID.Bytes(),
        a.CustomerID.Bytes(),
        a.AIcallID.Bytes(),
        a.AIID.Bytes(),
        a.PromptHistoryID.Bytes(),
        a.Language,
        a.TMCreate,
    )
    if err != nil {
        return 0, fmt.Errorf("AIAuditUpsert: could not execute. err: %v", err)
    }

    n, err := result.RowsAffected()
    if err != nil {
        return 0, fmt.Errorf("AIAuditUpsert: could not get rows affected. err: %v", err)
    }

    return n, nil
}

// aiauditGetFromDB loads an audit record directly from the DB.
func (h *handler) aiauditGetFromDB(id uuid.UUID) (*aiaudit.AIAudit, error) {
    cols := commondatabasehandler.GetDBFields(aiaudit.AIAudit{})

    query, args, err := sq.Select(cols...).
        From(aiauditTable).
        Where(sq.Eq{"id": id.Bytes()}).
        ToSql()
    if err != nil {
        return nil, fmt.Errorf("aiauditGetFromDB: could not build query. err: %v", err)
    }

    rows, err := h.db.Query(query, args...)
    if err != nil {
        return nil, fmt.Errorf("aiauditGetFromDB: could not query. err: %v", err)
    }
    defer func() { _ = rows.Close() }()

    if !rows.Next() {
        return nil, ErrNotFound
    }

    res := &aiaudit.AIAudit{}
    if err := commondatabasehandler.ScanRow(rows, res); err != nil {
        return nil, fmt.Errorf("aiauditGetFromDB: could not scan. err: %v", err)
    }

    return res, nil
}

// AIAuditGet returns an audit record by ID.
func (h *handler) AIAuditGet(ctx context.Context, id uuid.UUID) (*aiaudit.AIAudit, error) {
    return h.aiauditGetFromDB(id)
}

// AIAuditList returns a paginated list of audit records.
func (h *handler) AIAuditList(ctx context.Context, size uint64, token string, filters map[aiaudit.Field]any) ([]*aiaudit.AIAudit, error) {
    if token == "" {
        token = h.utilHandler.TimeGetCurTime()
    }

    cols := commondatabasehandler.GetDBFields(aiaudit.AIAudit{})

    builder := sq.Select(cols...).
        From(aiauditTable).
        Where(sq.Lt{"tm_create": token}).
        OrderBy("tm_create desc").
        Limit(size)

    builder, err := commondatabasehandler.ApplyFields(builder, filters)
    if err != nil {
        return nil, fmt.Errorf("AIAuditList: could not apply filters. err: %v", err)
    }

    query, args, err := builder.ToSql()
    if err != nil {
        return nil, fmt.Errorf("AIAuditList: could not build query. err: %v", err)
    }

    rows, err := h.db.Query(query, args...)
    if err != nil {
        return nil, fmt.Errorf("AIAuditList: could not query. err: %v", err)
    }
    defer func() { _ = rows.Close() }()

    res := []*aiaudit.AIAudit{}
    for rows.Next() {
        a := &aiaudit.AIAudit{}
        if err := commondatabasehandler.ScanRow(rows, a); err != nil {
            return nil, fmt.Errorf("AIAuditList: could not scan. err: %v", err)
        }
        res = append(res, a)
    }

    return res, nil
}

// AIAuditDelete soft-deletes an audit record.
func (h *handler) AIAuditDelete(ctx context.Context, id uuid.UUID) error {

// AIAuditUpdateFinal writes the goroutine's final result atomically.
// Only updates rows where status='progressing' AND tm_delete='9999-01-01' to
// prevent overwriting a 'failed' status set by the stale recovery sweep.
// Returns rowsAffected: 0 means the record was already soft-deleted or swept.
func (h *handler) AIAuditUpdateFinal(ctx context.Context, id uuid.UUID, status aiaudit.Status, overallScore *int, evaluation json.RawMessage, errStr string) (int64, error) {
    ts := h.utilHandler.TimeNow()

    var evalJSON sql.NullString
    if evaluation != nil {
        evalJSON = sql.NullString{String: string(evaluation), Valid: true}
    }

    query := fmt.Sprintf(`
        UPDATE %s
        SET status = ?, overall_score = ?, evaluation = ?, error = ?, tm_update = ?
        WHERE id = ? AND tm_delete = '%s' AND status = 'progressing'
    `, aiauditTable, aiauditSoftDeleteEpoch)

    result, err := h.db.ExecContext(ctx, query,
        string(status),
        overallScore,
        evalJSON,
        errStr,
        ts,
        id.Bytes(),
    )
    if err != nil {
        return 0, fmt.Errorf("AIAuditUpdateFinal: could not execute. err: %v", err)
    }

    n, err := result.RowsAffected()
    if err != nil {
        return 0, fmt.Errorf("AIAuditUpdateFinal: could not get rows affected. err: %v", err)
    }

    return n, nil
}
```

```go
// ParseEvaluationResponse parses and validates Gemini's JSON response.
// Returns an error if required fields are missing, scores are out of range,
// or score values are fractional (e.g., 4.5 is rejected).
func (h *geminiAuditHandler) ParseEvaluationResponse(data []byte) (*EvaluationResponse, error) {
    // Use float64 for initial parse to detect fractional scores.
    var raw struct {
        OverallScore float64 `json:"overall_score"`
        Dimensions   struct {
            Helpfulness    struct{ Score float64; Reason string } `json:"helpfulness"`
            Accuracy       struct{ Score float64; Reason string } `json:"accuracy"`
            Tone           struct{ Score float64; Reason string } `json:"tone"`
            GoalCompletion struct{ Score float64; Reason string } `json:"goal_completion"`
            ToolUsage      *struct{ Score float64; Reason string } `json:"tool_usage"`
        } `json:"dimensions"`
        Summary string `json:"summary"`
    }

    if err := json.Unmarshal(data, &raw); err != nil {
        return nil, fmt.Errorf("invalid JSON: %w", err)
    }

    validateScore := func(name string, v float64) error {
        if v != math.Trunc(v) {
            return fmt.Errorf("%s score is fractional: %v", name, v)
        }
        i := int(v)
        if i < 1 || i > 5 {
            return fmt.Errorf("%s score out of range [1,5]: %d", name, i)
        }
        return nil
    }

    if err := validateScore("overall_score", raw.OverallScore); err != nil {
        return nil, err
    }
    for _, dim := range []struct {
        name string
        v    float64
    }{
        {"helpfulness", raw.Dimensions.Helpfulness.Score},
        {"accuracy", raw.Dimensions.Accuracy.Score},
        {"tone", raw.Dimensions.Tone.Score},
        {"goal_completion", raw.Dimensions.GoalCompletion.Score},
    } {
        if err := validateScore(dim.name, dim.v); err != nil {
            return nil, err
        }
    }
    if raw.Dimensions.ToolUsage != nil {
        if err := validateScore("tool_usage", raw.Dimensions.ToolUsage.Score); err != nil {
            return nil, err
        }
    }

    const maxReason  = 2000
    const maxSummary = 5000

    checkLen := func(name, val string, max int) error {
        if len(val) > max {
            return fmt.Errorf("%s exceeds max length %d", name, max)
        }
        return nil
    }

    if err := checkLen("summary", raw.Summary, maxSummary); err != nil {
        return nil, err
    }
    for _, r := range []struct{ name, val string }{
        {"helpfulness.reason", raw.Dimensions.Helpfulness.Reason},
        {"accuracy.reason", raw.Dimensions.Accuracy.Reason},
        {"tone.reason", raw.Dimensions.Tone.Reason},
        {"goal_completion.reason", raw.Dimensions.GoalCompletion.Reason},
    } {
        if err := checkLen(r.name, r.val, maxReason); err != nil {
            return nil, err
        }
    }
    if raw.Dimensions.ToolUsage != nil {
        if err := checkLen("tool_usage.reason", raw.Dimensions.ToolUsage.Reason, maxReason); err != nil {
            return nil, err
        }
    }

    resp := &EvaluationResponse{
        OverallScore: int(raw.OverallScore),
        Summary:      raw.Summary,
        Dimensions: EvaluationDimensions{
            Helpfulness:    EvaluationDimension{Score: int(raw.Dimensions.Helpfulness.Score), Reason: raw.Dimensions.Helpfulness.Reason},
            Accuracy:       EvaluationDimension{Score: int(raw.Dimensions.Accuracy.Score), Reason: raw.Dimensions.Accuracy.Reason},
            Tone:           EvaluationDimension{Score: int(raw.Dimensions.Tone.Score), Reason: raw.Dimensions.Tone.Reason},
            GoalCompletion: EvaluationDimension{Score: int(raw.Dimensions.GoalCompletion.Score), Reason: raw.Dimensions.GoalCompletion.Reason},
        },
    }
    if raw.Dimensions.ToolUsage != nil {
        du := EvaluationDimension{Score: int(raw.Dimensions.ToolUsage.Score), Reason: raw.Dimensions.ToolUsage.Reason}
        resp.Dimensions.ToolUsage = &du
    }

    return resp, nil
}

// Evaluate runs the full evaluation: builds prompt, calls Gemini, parses response.
// Retries once on invalid JSON before returning error.
func (h *geminiAuditHandler) Evaluate(ctx context.Context, promptText, transcript, language string, hasTools bool) (*EvaluationResponse, json.RawMessage, error) {
    fullPrompt := h.BuildPrompt(promptText, transcript, language, false)

    var lastRaw []byte
    for attempt := 0; attempt < 2; attempt++ {
        resp, err := h.client.CreateChatCompletion(ctx, openai.ChatCompletionRequest{
            Model: geminiModel,
            Messages: []openai.ChatCompletionMessage{
                {Role: openai.ChatMessageRoleUser, Content: fullPrompt},
            },
        })
        if err != nil {
            return nil, nil, fmt.Errorf("gemini API error: %w", err)
        }

        if len(resp.Choices) == 0 {
            return nil, nil, fmt.Errorf("gemini returned no choices")
        }

        lastRaw = []byte(resp.Choices[0].Message.Content)
        parsed, parseErr := h.ParseEvaluationResponse(lastRaw)
        if parseErr == nil {
            return parsed, lastRaw, nil
        }
        // retry on parse error
    }

    return nil, nil, fmt.Errorf("invalid_evaluator_response after retry")
}
```

```go
// ValidateLanguage returns an error if lang is present but invalid BCP47 format.

// Create implements AIAuditHandler.
func (h *aiauditHandler) Create(ctx context.Context, customerID uuid.UUID, aicallID uuid.UUID, requestLanguage string) ([]*aiaudit.AIAudit, error) {
    // 1. Load aicall, verify owner, verify terminated
    ac, err := h.db.AIcallGet(ctx, aicallID)
    if err != nil {
        return nil, fmt.Errorf("%w", dbhandler.ErrNotFound)
    }
    if ac.CustomerID != customerID {
        return nil, fmt.Errorf("%w", dbhandler.ErrNotFound)
    }
    if ac.Status != aicall.StatusTerminated {
        return nil, fmt.Errorf("aicall not terminated")
    }

    // 2. Rate limit: count in-flight audits for this customer
    count, err := h.db.AIAuditCountProgressing(ctx, customerID)
    if err != nil {
        return nil, fmt.Errorf("could not count progressing audits: %w", err)
    }
    if count >= maxConcurrentCustomer {
        return nil, fmt.Errorf("rate limit exceeded")
    }

    // 3. Resolve language
    language := resolveLanguage(requestLanguage, ac.STTLanguage)

    // 4. Collect participant AI IDs
    var aiIDs []uuid.UUID
    if ac.AssistanceType == aicall.AssistanceTypeTeam {
        participants, err := h.db.ParticipantListByAIcallID(ctx, aicallID, 100, "")
        if err != nil {
            return nil, fmt.Errorf("could not list participants: %w", err)
        }
        for _, p := range participants {
            aiIDs = append(aiIDs, p.AIID)
        }
    } else {
        aiIDs = []uuid.UUID{ac.AssistanceID}
    }

    // 5. Upsert one audit record per AI; collect results
    var records []*aiaudit.AIAudit
    for _, aiID := range aiIDs {
        a := &aiaudit.AIAudit{
            AIcallID:    aicallID,
            AIID:        aiID,
            Language:    language,
            Status:      aiaudit.StatusProgressing,

        a.Identity.CustomerID = customerID

        // prompt_history_id resolved inside goroutine from metadata snapshot
        a.PromptHistoryID = uuid.Nil

        rowsAffected, err := h.db.AIAuditUpsert(ctx, a)
        if err != nil {
            return nil, fmt.Errorf("could not upsert audit: %w", err)
        }
        if rowsAffected == 0 {
            return nil, fmt.Errorf("audit already progressing") // caller returns 409
        }

        // Re-fetch to get the stable ID (might be existing row's ID on re-run)
        loaded, err := h.loadAuditByAIcallAndAIID(ctx, aicallID, aiID)
        if err != nil {
            return nil, fmt.Errorf("could not reload audit: %w", err)
        }
        records = append(records, loaded)

        // Spawn goroutine
        go h.runAuditJob(context.Background(), loaded.ID, ac, aiID, language)
    }

    return records, nil
}

// loadAuditByAIcallAndAIID fetches by aicall+ai (no cache, direct DB).
func (h *aiauditHandler) loadAuditByAIcallAndAIID(ctx context.Context, aicallID, aiID uuid.UUID) (*aiaudit.AIAudit, error) {
    filters := map[aiaudit.Field]any{
        aiaudit.FieldAIcallID: aicallID,
        aiaudit.FieldAIID:     aiID,
        aiaudit.FieldDeleted:  false,
    }
    list, err := h.db.AIAuditList(ctx, 1, "", filters)
    if err != nil || len(list) == 0 {
        return nil, fmt.Errorf("not found")
    }
    return list[0], nil
}

// runAuditJob is the per-AI background goroutine.
func (h *aiauditHandler) runAuditJob(ctx context.Context, auditID uuid.UUID, ac *aicall.AIcall, aiID uuid.UUID, language string) {
    log := logrus.WithFields(logrus.Fields{
        "func":     "runAuditJob",
        "audit_id": auditID,
        "ai_id":    aiID,
    })

    // Acquire global semaphore BEFORE starting the Gemini timeout
    h.semaphore <- struct{}{}
    defer func() { <-h.semaphore }()

    geminiCtx, cancel := context.WithTimeout(ctx, geminiTimeoutSeconds*time.Second)
    defer cancel()

    finalStatus := aiaudit.StatusFailed
    var finalScore *int
    var finalEvalJSON json.RawMessage
    var finalErr string

    defer func() {
        n, err := h.db.AIAuditUpdateFinal(context.Background(), auditID, finalStatus, finalScore, finalEvalJSON, finalErr)
        if err != nil {
            log.Errorf("AIAuditUpdateFinal failed: %v", err)
        }
        if n == 0 {
            log.Debugf("AIAuditUpdateFinal: no rows updated (already soft-deleted or swept)")
        }
    }()

    // Step 1: validate prompt_snapshots metadata
    snapshots, err := extractPromptSnapshots(ac)
    if err != nil {
        finalErr = string(aiaudit.ErrorInvalidCallMetadata)
        return
    }

    var promptText string
    var promptHistoryID uuid.UUID
    found := false
    for _, snap := range snapshots {
        if snap.AIID == aiID {
            promptText = snap.Prompt
            promptHistoryID = snap.PromptHistoryID
            found = true
            break
        }
    }
    if !found {
        finalErr = string(aiaudit.ErrorPromptSnapshotNotFound)
        return
    }
    if promptHistoryID == uuid.Nil {
        finalErr = string(aiaudit.ErrorPromptSnapshotNoHistoryID)
        return
    }

    // Update prompt_history_id on the audit record (best-effort, no retry)
    _ = h.db.AIAuditUpdateFinal(context.Background(), auditID, aiaudit.StatusProgressing, nil, nil, "")

    // Step 2: load messages
    var filters map[message.Field]any
    if ac.AssistanceType == aicall.AssistanceTypeTeam {
        filters = map[message.Field]any{
            message.FieldAIcallID:   ac.Identity.ID,
            message.FieldActiveAIID: aiID,
        }
    } else {
        filters = map[message.Field]any{
            message.FieldAIcallID: ac.Identity.ID,
        }
    }

    msgs, err := h.db.MessageList(geminiCtx, maxMessages, "", filters)
    if err != nil {
        log.Errorf("could not load messages: %v", err)
        finalErr = string(aiaudit.ErrorEvaluatorUnavailable)
        return
    }

    // Step 3: check context cancellation
    select {
    case <-geminiCtx.Done():
        finalErr = string(aiaudit.ErrorCancelled)
        return
    default:
    }

    // Step 4: build transcript (content already sanitized inside BuildPrompt)
    truncated := false
    transcript := buildTranscript(msgs, &truncated)

    // Step 5 (language already resolved by Create)

    // Step 6+7: call Gemini
    result, rawJSON, err := h.geminiHandler.Evaluate(geminiCtx, promptText, transcript, language, hasToolCalls(msgs))
    if err != nil {
        log.Errorf("Gemini evaluation failed: %v", err)
        if strings.Contains(err.Error(), "invalid_evaluator_response") {
            finalErr = string(aiaudit.ErrorInvalidEvaluatorResponse)
        } else {
            finalErr = string(aiaudit.ErrorEvaluatorUnavailable)
        }
        return
    }

    // Step 8: write final result (in defer above)
    score := result.OverallScore
    finalScore = &score
    finalEvalJSON = rawJSON
    finalStatus = aiaudit.StatusCompleted
    finalErr = ""

    _ = promptHistoryID // used to update prompt_history_id; handled in a real implementation
}

// extractPromptSnapshots parses the metadata field for prompt snapshots.
func extractPromptSnapshots(ac *aicall.AIcall) ([]aicall.PromptSnapshot, error) {
    raw, ok := ac.Metadata[aicall.MetaKeyPromptSnapshots]
    if !ok || raw == nil {
        return nil, fmt.Errorf("missing prompt_snapshots")
    }

    data, err := json.Marshal(raw)
    if err != nil {
        return nil, fmt.Errorf("could not marshal prompt_snapshots: %w", err)
    }

    var snaps []aicall.PromptSnapshot
    if err := json.Unmarshal(data, &snaps); err != nil {
        return nil, fmt.Errorf("could not unmarshal prompt_snapshots: %w", err)
    }

    return snaps, nil
}

// buildTranscript formats message list into a string for the Gemini prompt.
// Truncates each message content to maxMessageContentLen characters.
func buildTranscript(msgs []*message.Message, truncated *bool) string {
    var sb strings.Builder
    if len(msgs) >= maxMessages {
        *truncated = true
    }
    for _, m := range msgs {
        content := m.Content
        if len(content) > maxMessageContentLen {
            content = content[:maxMessageContentLen]
        }
        role := string(m.Role)
        if len(m.ToolCalls) > 0 {
            tc, _ := json.Marshal(m.ToolCalls)
            sb.WriteString(fmt.Sprintf("[tool_call]: %s\n", string(tc)))
        } else if m.ToolCallID != "" {
            sb.WriteString(fmt.Sprintf("[tool_result]: %s\n", content))
        } else {
            sb.WriteString(fmt.Sprintf("[%s]: %s\n", role, content))
        }
    }
    return sb.String()
}

// hasToolCalls returns true if any message has tool calls.
func hasToolCalls(msgs []*message.Message) bool {
    for _, m := range msgs {
        if len(m.ToolCalls) > 0 || m.ToolCallID != "" {
            return true
        }
    }
    return false
}
```

```go
// AIAuditCreate triggers an audit for the given aicall.
func (h *serviceHandler) AIAuditCreate(
    ctx context.Context,
    a *auth.AuthIdentity,
    aicallID uuid.UUID,
    language string,

    if !h.hasPermission(ctx, a, a.CustomerID, amagent.PermissionCustomerAdmin|amagent.PermissionCustomerManager) {
        return nil, serviceerrors.ErrPermissionDenied
    }

    if language != "" && !bcp47Re.MatchString(language) {
        return nil, errors.Wrapf(serviceerrors.ErrInvalidArgument, "invalid BCP47 language: %s", language)
    }

    records, err := h.reqHandler.AIV1AIAuditCreate(ctx, a.CustomerID, aicallID, language)

        return nil, errors.Wrapf(err, "could not create aiaudit")
    }

    return res, nil
}

// AIAuditGetsByCustomerID returns a paginated list of audit records.
func (h *serviceHandler) AIAuditGetsByCustomerID(
    ctx context.Context,
    a *auth.AuthIdentity,
    size uint64,
    token string,
    aicallID, aiID uuid.UUID,
) ([]*amaiaudit.WebhookMessage, error) {

        return nil, serviceerrors.ErrPermissionDenied
    }

    if token == "" {
        token = h.utilHandler.TimeGetCurTime()
    }

    filters := map[string]any{
        "deleted":     "false",
        "customer_id": a.CustomerID.String(),
    }
    if aicallID != uuid.Nil {
        filters["aicall_id"] = aicallID.String()
    }
    if aiID != uuid.Nil {
        filters["ai_id"] = aiID.String()
    }

    typedFilters, err := h.convertAIAuditFilters(filters)
    if err != nil {

    records, err := h.reqHandler.AIV1AIAuditList(ctx, token, size, typedFilters)
    if err != nil {

    res := make([]*amaiaudit.WebhookMessage, 0, len(records))
```

```rst
.. _ai-aiaudit-overview:

AI Audit Overview
=================

AI audits evaluate the performance of an AI participant in a completed AI call.
When triggered, VoIPbin calls Gemini to score the AI on helpfulness, accuracy, tone,
goal completion, and tool usage. Results are stored and queryable via API.

Endpoints
---------

- ``POST /aiaudits`` — trigger an audit for a completed aicall
- ``GET /aiaudits`` — list audits (filterable by ``aicall_id`` and ``ai_id``)
- ``GET /aiaudits/{id}`` — get a single audit by ID
- ``DELETE /aiaudits/{id}`` — delete an audit record

Usage Flow
----------

1. An aicall must have ``status = "terminated"`` before it can be audited.
2. ``POST /aiaudits`` returns ``202 Accepted`` immediately and triggers one background evaluation per AI participant.
3. Poll ``GET /aiaudits/{id}`` until ``status`` is ``"completed"`` or ``"failed"``.
4. On ``"completed"``, ``overall_score`` (1–5) and the full ``evaluation`` object are available.
5. Re-running ``POST /aiaudits`` for the same call resets and re-runs the evaluation.

Each AI participant in a team call gets its own audit record. For single-AI calls,
one audit record is created.
```
