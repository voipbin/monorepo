## Task 2: Create service taxonomy script and generate taxonomy manifest

**Files:**
- Create: `docs/reference/service-taxonomy-gen.sh`
- Create: `docs/reference/service-taxonomy.md`

**Step 1: Create the taxonomy script**

Create `docs/reference/service-taxonomy-gen.sh`:
```bash
#!/usr/bin/env bash
# Generates service class assignments from filesystem ground truth.
# Order matters — each check eliminates by precedence.
# Run from monorepo root: bash docs/reference/service-taxonomy-gen.sh

set -euo pipefail

echo "# Service Taxonomy"
echo "# Generated by docs/reference/service-taxonomy-gen.sh"
echo "# DO NOT EDIT MANUALLY — regenerate and commit when services change"
echo ""
echo "| Service | Class | Notes |"
echo "|---------|-------|-------|"

for svc_dir in bin-*/  voip-*/; do
  name=$(basename "$svc_dir")

  # Class D: Python/Alembic (alembic.ini present)
  if [ -f "${svc_dir}alembic.ini" ]; then
    echo "| $name | D | Python/Alembic migrations |"
    continue
  fi

  # Class E: OpenAPI codegen (no cmd/, has openapi/)
  if [ ! -d "${svc_dir}cmd" ] && [ -d "${svc_dir}openapi" ]; then
    echo "| $name | E | OpenAPI codegen, no runtime |"
    continue
  fi

  # Class C: shared library (no cmd/, has pkg/)
  if [ ! -d "${svc_dir}cmd" ] && [ -d "${svc_dir}pkg" ]; then
    echo "| $name | C | Shared library, no cmd/ |"
    continue
  fi

  # Class B: API gateway (has cmd/, no listenhandler, uses oapi-codegen)
  if [ -d "${svc_dir}cmd" ] && [ ! -d "${svc_dir}pkg/listenhandler" ] && \
     grep -q "oapi-codegen" "${svc_dir}go.mod" 2>/dev/null; then
    echo "| $name | B | HTTP/REST gateway |"
    continue
  fi

  # Class A2: event-driven worker (has cmd/, no listenhandler, not B)
  if [ -d "${svc_dir}cmd" ] && [ ! -d "${svc_dir}pkg/listenhandler" ]; then
    echo "| $name | A2 | Event-driven worker, no inbound RPC |"
    continue
  fi

  # Class A+sub: voip-* proxies with listenhandler
  if [ -d "${svc_dir}pkg/listenhandler" ] && [[ "$name" == voip-* ]]; then
    echo "| $name | A+sub | Go RPC + embedded native daemon |"
    continue
  fi

  # Class A: standard Go RPC manager
  if [ -d "${svc_dir}pkg/listenhandler" ]; then
    echo "| $name | A | Standard Go RPC manager |"
    continue
  fi

  echo "| $name | UNKNOWN | Manual classification needed |"
done
```

## Task 4: Build and validate the extractor script

**Files:**
- Create: `docs/reference/extractor.sh`

**Step 1: Create the extractor script**

Create `docs/reference/extractor.sh`:
```bash
#!/usr/bin/env bash
# Extracts structured data from a Go service for doc generation.
# Usage: bash docs/reference/extractor.sh <service-dir>
# Output: docs/.docs-gen/<service-name>.extracted.json

set -euo pipefail

SVC_DIR="${1:?Usage: $0 <service-dir>}"
SVC_NAME=$(basename "$SVC_DIR")
OUT_DIR="docs/.docs-gen"
OUT_FILE="$OUT_DIR/${SVC_NAME}.extracted.json"

mkdir -p "$OUT_DIR"

# Determine class from taxonomy
CLASS=$(grep "^| $SVC_NAME |" docs/reference/service-taxonomy.md | awk -F'|' '{print $3}' | xargs)

echo "Extracting $SVC_NAME (Class: $CLASS)" >&2

# --- Routing table (listenhandler) ---
ROUTING="[]"
if [ -d "$SVC_DIR/pkg/listenhandler" ]; then
  ROUTING=$(grep -n 'regexp\.MustCompile' "$SVC_DIR/pkg/listenhandler/main.go" 2>/dev/null \
    | sed 's/.*regexp\.MustCompile(`\?\([^`"]*\)`\?").*/\1/' \
    | jq -R '{pattern: .}' | jq -s '.' 2>/dev/null || echo "[]")
fi

# --- Events subscribed (cmd/*/main.go) ---
EVENTS_SUB="[]"
MAIN_GO=$(ls "$SVC_DIR"/cmd/*/main.go 2>/dev/null | head -1 || true)
if [ -n "$MAIN_GO" ]; then
  EVENTS_SUB=$(awk '/subscribeTargets.*:?=.*\[\]string\{/,/\}/' "$MAIN_GO" 2>/dev/null \
    | grep -oP '"[^"]*"' | tr -d '"' \
    | jq -R '{queue_symbol: .}' | jq -s '.' 2>/dev/null || echo "[]")
fi

# --- Events published (pkg/**/go + internal/**/*.go) ---
EVENTS_PUB="[]"
if [ -d "$SVC_DIR/pkg" ]; then
  EVENTS_PUB=$(find "$SVC_DIR/pkg" "$SVC_DIR/internal" -name "*.go" \
    ! -name "*_test.go" ! -name "mock_*.go" 2>/dev/null \
    | xargs grep -l 'PublishWebhookEvent\|CallPublishEvent' 2>/dev/null \
    | xargs grep -hn 'PublishWebhookEvent\|CallPublishEvent' 2>/dev/null \
    | grep -oP '(call|flow|agent|billing|ai|message|email|conference|queue|campaign|contact|conversation|customer|number|outdial|rag|registrar|route|sentinel|hook|storage|tag|talk|timeline|transcribe|transfer|tts|webhook)\.[A-Za-z]+' \
    | sort -u \
    | jq -R '{event_type_symbol: .}' | jq -s '.' 2>/dev/null || echo "[]")
fi

# --- Dependencies (go.mod replace) ---
DEPS="[]"
if [ -f "$SVC_DIR/go.mod" ]; then
  DEPS=$(grep '^replace ' "$SVC_DIR/go.mod" 2>/dev/null \
    | sed 's/replace \(.*\) => \(.*\)/\1|\2/' \
    | jq -R 'split("|") | {module_path: .[0], local_path: .[1]}' \
    | jq -s '.' 2>/dev/null || echo "[]")
fi

# --- Config vars (internal/config/*.go or cmd/*/init.go) ---
CONFIG="[]"
CONFIG_SRC=$(ls "$SVC_DIR"/internal/config/*.go 2>/dev/null | head -1 \
  || ls "$SVC_DIR"/cmd/*/init.go 2>/dev/null | head -1 || true)
if [ -n "$CONFIG_SRC" ]; then
  CONFIG=$(grep -n 'f\.String\|f\.Int\|f\.Bool\|f\.StringSlice' "$CONFIG_SRC" 2>/dev/null \
    | grep -oP '"[a-z_]+"' | tr -d '"' \
    | jq -R '{flag: .}' | jq -s '.' 2>/dev/null || echo "[]")
fi

# --- Prometheus metrics ---
METRICS="[]"
if [ -d "$SVC_DIR/pkg" ]; then
  METRICS=$(grep -rn \
    'prometheus\.NewHistogramVec\|prometheus\.NewCounterVec\|prometheus\.NewGaugeVec\|prometheus\.NewSummaryVec' \
    "$SVC_DIR" --include="*.go" 2>/dev/null \
    | grep -oP '"[a-z_]+"' | head -1 | tr -d '"' \
    | jq -R '{name: ., type: "histogram"}' | jq -s '.' 2>/dev/null || echo "[]")
fi

# --- Missing fields ---
MISSING="[]"
[ "$ROUTING" = "[]" ] && [ "$CLASS" = "A" ] && MISSING=$(echo "$MISSING" | jq '. + ["routing_table"]')
[ "$EVENTS_SUB" = "[]" ] && MISSING=$(echo "$MISSING" | jq '. + ["events_subscribed"]')
[ "$CONFIG" = "[]" ] && MISSING=$(echo "$MISSING" | jq '. + ["config_vars"]')

# --- Assemble JSON ---
jq -n \
  --arg schema_version "1.0" \
  --arg service_name "$SVC_NAME" \
  --arg service_class "$CLASS" \
  --argjson routing_table "$ROUTING" \
  --argjson events_subscribed "$EVENTS_SUB" \
  --argjson events_published "$EVENTS_PUB" \
  --argjson dependencies "$DEPS" \
  --argjson config_vars "$CONFIG" \
  --argjson metrics "$METRICS" \
  --argjson missing_fields "$MISSING" \
  '{
    schema_version: $schema_version,
    service_name: $service_name,
    service_class: $service_class,
    routing_table: $routing_table,
    events_subscribed: $events_subscribed,
    events_published: $events_published,
    dependencies: $dependencies,
    config_vars: $config_vars,
    metrics: $metrics,
    missing_fields: $missing_fields
  }' > "$OUT_FILE"

echo "Written: $OUT_FILE" >&2
```
